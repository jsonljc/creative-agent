# PCD Identity Registry — SP1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Land the data foundation for the PCD identity registry — Prisma models, Zod schemas, stores, and a backfill job that promotes every existing PCD job into Tier 1 — with zero behavior change to live PCD execution.

**Architecture:** Extend the existing `CreatorIdentity` and `CreativeJob` models in place (no fork). Add four new models — `ProductIdentity`, `ProductImage`, `ConsentRecord`, `ProductQcResult` — plus a `PcdIdentitySnapshot` table that hangs off `AssetRecord` for per-generation lineage. New Zod contracts in `@switchboard/schemas`. New Prisma stores in `@switchboard/db`. Backfill runs as an Inngest job in `apps/api`. No imports into `packages/core`. No changes to PCD orchestration, providers, or UI in this sprint.

**Tech Stack:** TypeScript (ESM), pnpm + Turborepo, Prisma + PostgreSQL, Zod, Inngest, Vitest.

**Spec:** `05_OUTPUTS/Plans/2026-04-27-pcd-identity-registry-design.md`

**Repo:** `~/switchboard`

**Working invariants** (verify at the start of each session):
- All commands run from `~/switchboard` unless noted.
- Conventional Commits enforced. Use `feat(pcd-registry): ...` prefix.
- Every new module includes co-located tests (`*.test.ts` next to the source, or under `__tests__/`).
- Run `pnpm typecheck` and `pnpm test` before each commit.
- Schema changes require a Prisma migration in the **same commit**.
- Dependency layers must hold: `schemas` → no `@switchboard/*` imports; `db` → `schemas + core` only; never import `db` from `core` or `creative-pipeline`.

```ts
import type { PrismaClient } from "@prisma/client";
import {

  type ProductImage,
  type ProductImageViewType,
  type ProductQualityTier,
  type ProductLockStatus,

export interface CreateProductIdentityInput {
  orgId: string;
  title: string;
  sourceUrl?: string;
  description?: string;
  brandName?: string;
  sku?: string;
  packageType?: string;
  canonicalPackageText?: string;
  qualityTier?: ProductQualityTier;
  lockStatus?: ProductLockStatus;

  viewType: ProductImageViewType;
  uri: string;
  approvedForGeneration: boolean;
  hasReadableLabel?: boolean;
  ocrText?: string;
  backgroundType?: string;
}

export class PrismaProductIdentityStore {
  constructor(private readonly prisma: PrismaClient) {}

  async create(input: CreateProductIdentityInput): Promise<ProductIdentity> {
    const row = await this.prisma.productIdentity.create({
      data: {

        title: input.title,
        sourceUrl: input.sourceUrl,
        description: input.description,

        lockStatus: input.lockStatus ?? "draft",
      },
    });
    return ProductIdentitySchema.parse(row);
  }

  async getById(id: string): Promise<ProductIdentity | null> {
    const row = await this.prisma.productIdentity.findUnique({ where: { id } });
    return row ? ProductIdentitySchema.parse(row) : null;
  }

  async addImage(productIdentityId: string, input: AddProductImageInput): Promise<ProductImage> {
    const row = await this.prisma.productImage.create({
      data: {
        productIdentityId,
        viewType: input.viewType,
        uri: input.uri,
        approvedForGeneration: input.approvedForGeneration,
        hasReadableLabel: input.hasReadableLabel,
        ocrText: input.ocrText,
        backgroundType: input.backgroundType,
      },
    });
    return ProductImageSchema.parse(row);
  }

  async listImages(productIdentityId: string): Promise<ProductImage[]> {
    const rows = await this.prisma.productImage.findMany({
      where: { productIdentityId },
      orderBy: { createdAt: "asc" },
    });

  }
```

```ts
        revocable: input.revocable ?? true,
        revoked: false,
        recordingUri: input.recordingUri,
        effectiveAt: input.effectiveAt,
        expiresAt: input.expiresAt,
      },
    });
    return ConsentRecordSchema.parse(row);
```

```ts
// apps/api/src/services/cron/pcd-registry-backfill.ts
import { inngest } from "../../bootstrap/inngest.js";
import { prisma } from "../../bootstrap/prisma.js";
import { PrismaProductIdentityStore } from "@switchboard/db";
import { PrismaCreatorIdentityStore } from "@switchboard/db";
import { PrismaCreativeJobStore } from "@switchboard/db";

export interface BackfillJobInput {
  id: string;
  organizationId: string;
  deploymentId: string;
  productDescription: string;
  productImages: string[];
  registryBackfilled?: boolean;
}

export interface BackfillStores {
  productStore: {
    findOrCreateForJob: (input: BackfillJobInput) => Promise<{ id: string }>;
  };
  creatorStore: {
    findOrCreateStockForDeployment: (deploymentId: string) => Promise<{ id: string }>;
  };
  jobStore: {
    markRegistryBackfilled: (
      jobId: string,
      input: { productIdentityId: string; creatorIdentityId: string },
    ) => Promise<{ id: string; registryBackfilled: boolean; productIdentityId: string; creatorIdentityId: string }>;
  };
}

export async function backfillJobOnce(
  job: BackfillJobInput,
  stores: BackfillStores,
) {
  if (job.registryBackfilled) {
    return { id: job.id, registryBackfilled: true };
  }
  const product = await stores.productStore.findOrCreateForJob(job);
  const creator = await stores.creatorStore.findOrCreateStockForDeployment(job.deploymentId);
  return stores.jobStore.markRegistryBackfilled(job.id, {
    productIdentityId: product.id,
    creatorIdentityId: creator.id,
  });
}

// Adapter wiring stores to real Prisma stores
function buildLiveStores(): BackfillStores {
  const product = new PrismaProductIdentityStore(prisma);
  const creator = new PrismaCreatorIdentityStore(prisma);
  const job = new PrismaCreativeJobStore(prisma);

  return {
    productStore: {
      findOrCreateForJob: async (j) => {
        const existing = await prisma.productIdentity.findFirst({
          where: { orgId: j.organizationId, title: j.productDescription },
        });
        if (existing) return { id: existing.id };
        return product.create({
          orgId: j.organizationId,
          title: j.productDescription,
          qualityTier: "url_imported",
          lockStatus: "draft",
        });
      },
    },
    creatorStore: {
      findOrCreateStockForDeployment: async (deploymentId) => {
        const existing = await prisma.creatorIdentity.findFirst({
          where: { deploymentId, qualityTier: "stock" },
        });
        if (existing) return { id: existing.id };
        const created = await prisma.creatorIdentity.create({
          data: {
            deploymentId,
            name: "Stock Avatar (backfill)",
            identityRefIds: [],
            heroImageAssetId: "backfill_placeholder",
            identityDescription: "Backfilled stock avatar (Tier 1).",
            voice: {
              voiceId: "stock",
              provider: "elevenlabs",
              tone: "neutral",
              pace: "moderate",
              sampleUrl: "",
            },
            personality: { energy: "conversational", deliveryStyle: "natural" },
            appearanceRules: { hairStates: [], wardrobePalette: [] },
            environmentSet: [],
            qualityTier: "stock",
          },
        });
        return { id: created.id };
      },
    },
    jobStore: {
      markRegistryBackfilled: (id, input) => job.markRegistryBackfilled(id, input),
    },
  };
}

export const pcdRegistryBackfill = inngest.createFunction(
  { id: "pcd-registry-backfill", retries: 3 },
  { event: "pcd/registry.backfill.requested" },
  async ({ step, event }) => {
    const stores = buildLiveStores();
    const batchSize = 50;
    const orgFilter = (event.data as { orgId?: string } | undefined)?.orgId;

    let processed = 0;
    while (true) {
      const batch = await step.run("fetch-batch", async () =>
        prisma.creativeJob.findMany({
          where: {
            registryBackfilled: false,
            ...(orgFilter ? { organizationId: orgFilter } : {}),
          },
          take: batchSize,
        }),
      );
      if (batch.length === 0) break;

      for (const job of batch) {
        await step.run(`backfill-${job.id}`, () =>
          backfillJobOnce(
            {
              id: job.id,
              organizationId: job.organizationId,
              deploymentId: job.deploymentId,
              productDescription: job.productDescription,
              productImages: job.productImages,
              registryBackfilled: job.registryBackfilled,
            },
            stores,
          ),
        );
        processed += 1;
      }
    }
    return { processed };
  },
);
```
